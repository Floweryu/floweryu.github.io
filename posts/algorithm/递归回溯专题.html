public void dfs(List<String> res, int n, StringBuilder sb, int open, int close) {
    if (sb.length() == 2 * n) {
        res.add(sb.toString());

    if (open < n) {

        // ç„¶ååˆè¿›å…¥dfsé€’å½’, è¿™æ—¶close < open, å°±ä¼šæ·»åŠ ), å³(()
        sb.deleteCharAt(sb.length() - 1);

    if (close < open) {

    private int rows, cols;
    public int longestIncreasingPath(int[][] matrix) {
        if (matrix == null || matrix.length == 0 || matrix[0].length == 0) {
            return 0;
        }
        rows = matrix.length;
        cols = matrix[0].length;
        // è®°å¿†çŸ©é˜µ: è®°å½•éå†å½“å‰ç‚¹ä½æ—¶æœ€å¤§è·¯å¾„
        int[][] memo = new int[rows][cols];
        int ans = 0;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                // éå†æ¯ä¸ªçŸ©é˜µç‚¹ä½
                ans = Math.max(ans, dfs(matrix, i, j, memo));
            }
        }
        return ans;
    }

    public int dfs(int[][] matrix, int i, int j, int[][] memo) {
        // å¦‚æœå½“å‰ç‚¹ä½å·²ç»éå†è¿‡ï¼Œè¿”å›å½“å‰ç‚¹ä½æ—¶è·¯å¾„é•¿åº¦
        if (memo[i][j] != 0) {
            return memo[i][j];

        // éå†åˆ°å½“å‰å®šä½, åˆ™è·¯å¾„+1
        // è¿™é‡Œå…ˆæŠŠå½“å‰è¿™ä¸ªç‚¹è·¯é•¿åº¦ç®—å‡ºæ¥ï¼Œæœ€çŸ­å°±åªåŒ…å«å½“å‰ç‚¹é•¿åº¦1ï¼Œä¸‹é¢é‚£ä¸ªdfsä¸­çš„+1ä¸ä¸€å®šèƒ½èµ°çš„åˆ°ï¼Œä¸æ»¡è¶³é€’å¢çš„è¯å°±ä¸èµ°é‚£é‡Œäº†ï¼Œè¿™æ ·å½“å‰ç‚¹å±€éƒ¨æœ€é•¿è·¯å¾„é•¿åº¦ä¾æ—§è¦èµ‹å€¼ä¸º1
        memo[i][j]++;
        // éå†å››ä¸ªæ–¹å‘
        for (int[] dir : dirs) {
            int newI = i + dir[0], newJ = j + dir[1];
            // 1. åˆ¤æ–­æ˜¯å¦è¶Šç•Œ
            // 2. åˆ¤æ–­ä¸‹ä¸€æ­¥éå†çš„ç‚¹ä½å€¼æ˜¯å¦æ¯”å½“å‰å€¼å¤§(è´ªå¿ƒ)

            }
        }
