public int[] mergeSort(int[] nums1, int[] nums2) {
    int len1 = nums1.length, len2 = nums2.length;
    if (len1 < 1) return nums2;
    if (len2 < 1) return nums1;
    int[] res = new int[len1 + len2];
    int i = 0, j = 0, index = 0;
    while (i < len1 && j < len2) {
        res[index++] = nums1[i] < nums2[j] ? nums1[i++] : nums2[j++];
    }
    while (i < len1) {
        res[index++] = nums1[i++];
    }
    while (j < len2) {
        res[index++] = nums2[j++];
    }
    return res;
}
