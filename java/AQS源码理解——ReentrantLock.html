private final boolean parkAndCheckInterrupt() {
    // å°†å½“å‰çº¿ç¨‹æŒ‚èµ·
    LockSupport.park(this);

}

/**
 * Cancels an ongoing attempt to acquire.
 * å–æ¶ˆæŒ‡å®šnodeå‚ä¸ç«äº‰
 * @param node the node
 */
private void cancelAcquire(Node node) {
    // Ignore if node doesn't exist
    if (node == null)
        return;

    // å› ä¸ºå·²ç»å–æ¶ˆæ’é˜Ÿäº†ï¼Œæ‰€ä»¥nodeå†…éƒ¨å…³è”çš„å½“å‰çº¿ç¨‹ç½®ä¸ºnullå°±å¥½äº†
    node.thread = null;

    // Skip cancelled predecessors
    // è·å–å½“å‰å–æ¶ˆæ’é˜Ÿnodeçš„å‰é©±
    Node pred = node.prev;
    // è·³è¿‡éœ€è¦éœ€è¦æ’é˜Ÿçš„èŠ‚ç‚¹
    while (pred.waitStatus > 0)
        node.prev = pred = pred.prev;

    // predNext is the apparent node to unsplice. CASes below will
    // fail if not, in which case, we lost race vs another cancel
    // or signal, so no further action is necessary.
    // æ‹¿åˆ°å‰é©±çš„nextèŠ‚ç‚¹, æœ‰ä¸€ä¸‹æƒ…å†µ
    // 1.å½“å‰node
    Node predNext = pred.next;

    // Can use unconditional write instead of CAS here.
    // After this atomic step, other Nodes can skip past us.
    // Before, we are free of interference from other threads.
    // å°†å½“å‰nodeçŠ¶æ€è®¾ç½®ä¸ºå–æ¶ˆçŠ¶æ€1
    node.waitStatus = Node.CANCELLED;

    // If we are the tail, remove ourselves.
    /**
         * å½“å‰å–æ¶ˆæ’é˜Ÿçš„nodeæ‰€åœ¨é˜Ÿåˆ—çš„ä½ç½®ä¸åŒï¼Œæ‰§è¡Œå‡ºé˜Ÿç­–ç•¥æ˜¯ä¸ä¸€æ ·çš„ï¼Œä¸€å…±åˆ†ä¸º3ç§æƒ…å†µã€‚
         * 1:å½“å‰nodeæ˜¯é˜Ÿå°¾ tail -> node
         * 2:å½“å‰nodeä¸æ˜¯head.nextèŠ‚ç‚¹ï¼Œä¹Ÿä¸æ˜¯tailèŠ‚ç‚¹
         * 3:å½“å‰nodeæ˜¯head.nextèŠ‚ç‚¹ã€‚
         */

    // ç¬¬ä¸€ç§æƒ…å†µ
    // æ¡ä»¶ä¸€: node=tailæˆç«‹ï¼Œå½“å‰nodeæ˜¯é˜Ÿå°¾ tail -> node
    // æ¡ä»¶äºŒ:compareAndSetTail(node,pred)æˆåŠŸçš„è¯, è¯´æ˜ä¿®æ”¹tailå®Œæˆ, å³å°†å½“å‰èŠ‚ç‚¹çš„å‰é©±è®¾ç½®ä¸ºtail
    if (node == tail && compareAndSetTail(node, pred)) {
        // ä¿®æ”¹pred.next -> null,å®Œæˆnodeå‡ºé˜Ÿ
        compareAndSetNext(pred, predNext, null);
    } else {
        // If successor needs signal, try to set pred's next-link
        // so it will get one. Otherwise wake it up to propagate.
        // ä¿å­˜èŠ‚ç‚¹çŠ¶æ€
        int ws;

        // ç¬¬äºŒç§æƒ…å†µï¼šå½“å‰nodeä¸æ˜¯head.nextèŠ‚ç‚¹ï¼Œä¹Ÿä¸æ˜¯tail
        // æ¡ä»¶ä¸€ï¼špred!=head è¯´æ˜å½“å‰nodeä¸æ˜¯head.nextèŠ‚ç‚¹ï¼Œä¹Ÿä¸æ˜¯tailèŠ‚ç‚¹
        // æ¡ä»¶äºŒï¼š ((ws = pred.waitStatus) == Node.SIGNAL || (ws <= 0 && compareAndSetWaitStatus(pred, ws, Node.SIGNAL)))
        // æ¡ä»¶2.1 (ws = pred.waitStatus) == Node.SIGNAL æˆç«‹:è¯´æ˜nodeçš„å‰é©±çŠ¶æ€æ˜¯SignalçŠ¶æ€. ä¸æˆç«‹ï¼šå‰é©±çŠ¶æ€å¯èƒ½æ˜¯0ï¼Œ
        // æç«¯æƒ…å†µä¸‹ï¼šå‰é©±ä¹Ÿå–æ¶ˆæ’é˜Ÿäº†
        // æ¡ä»¶2.2: (ws <= 0 && compareAndSetWaitStatus(pred, ws, Node.SIGNAL))
        // å‡è®¾å‰é©±çŠ¶æ€ ws<=0 åˆ™è®¾ç½®å‰é©±çŠ¶æ€ä¸ºSignalçŠ¶æ€ï¼Œè¡¨ç¤ºè¦å”¤é†’åç»§èŠ‚ç‚¹

        // ifé‡Œé¢åšçš„äº‹æƒ…ï¼Œå°±æ˜¯è®©pred.next -> node.next, æ‰€æœ‰éœ€è¦ä¿è¯predèŠ‚ç‚¹çŠ¶æ€ä¸ºSignalçŠ¶æ€
        if (pred != head &&
            ((ws = pred.waitStatus) == Node.SIGNAL ||
             (ws <= 0 && compareAndSetWaitStatus(pred, ws, Node.SIGNAL))) &&
            pred.thread != null) {

            // æƒ…å†µ2ï¼šå½“å‰nodeä¸æ˜¯head.nextèŠ‚ç‚¹ï¼Œä¹Ÿä¸æ˜¯tail
            // å‡ºé˜Ÿ: pred.next -> node.next èŠ‚ç‚¹åï¼Œå½“node.nextèŠ‚ç‚¹è¢«å”¤é†’åï¼Œè°ƒç”¨shouldParkAfterFailedAcquireä¼šè®©node.nextèŠ‚ç‚¹è¶Šè¿‡å–æ¶ˆçŠ¶æ€çš„èŠ‚ç‚¹ï¼Œå®ŒæˆçœŸæ­£çš„å‡ºé˜Ÿ
            Node next = node.next;
            if (next != null && next.waitStatus <= 0)
                compareAndSetNext(pred, predNext, next);
        } else {
            // å½“å‰nodeæ˜¯head.nextèŠ‚ç‚¹
            // ç±»ä¼¼æƒ…å†µ2ï¼Œåç»§èŠ‚ç‚¹å”¤é†’åï¼Œä¼šè°ƒç”¨shouldParkAfterFailedAcquireä¼šè®©node.nextèŠ‚ç‚¹è¶Šè¿‡å–æ¶ˆçŠ¶æ€çš„èŠ‚ç‚¹
            // é˜Ÿåˆ—çš„ç¬¬ä¸‰ä¸ªèŠ‚ç‚¹ä¼šç›´æ¥ä¸headå»ºç«‹åŒé‡æŒ‡å‘çš„å…³ç³»ï¼Œhead.next->ç¬¬ä¸‰ä¸ªnode ä¸­é—´å°±æ˜¯è¢«å‡ºé˜Ÿçš„head.nextèŠ‚ç‚¹ ç¬¬ä¸‰ä¸ªnode.prev -> head
            unparkSuccessor(node);
        }

        node.next = node; // help GC
    }
}

/**
 * Wakes up node's successor, if one exists.
 *
 * @param node the node
 */
private void unparkSuccessor(Node node) {
    /*
         * If status is negative (i.e., possibly needing signal) try
         * to clear in anticipation of signalling.  It is OK if this
         * fails or if status is changed by waiting thread.
         */
    // è·å–å½“å‰èŠ‚ç‚¹çš„çŠ¶æ€
    int ws = node.waitStatus;
    if (ws < 0)
        //-1 Signal æ”¹æˆ0çš„åŸå› ï¼šå› ä¸ºå½“å‰èŠ‚ç‚¹å·²ç»å®Œæˆå–Šåç»­èŠ‚ç‚¹çš„ä»»åŠ¡äº†ã€‚
        compareAndSetWaitStatus(node, ws, 0);

    /*
         * Thread to unpark is held in successor, which is normally
         * just the next node.  But if cancelled or apparently null,
         * traverse backwards from tail to find the actual
         * non-cancelled successor.
         */

    // æ¡ä»¶ä¸€ï¼š
    // sä»€ä¹ˆæ—¶å€™ç­‰äºnullï¼Ÿ
    // 1.å½“å‰èŠ‚ç‚¹å°±æ˜¯tailèŠ‚ç‚¹æ—¶ï¼Œs==null

    // æ¡ä»¶äºŒï¼š s.waitStatus>0 å‰æï¼šs!=null
    // æˆç«‹ï¼šè¯´æ˜å½“å‰nodeèŠ‚ç‚¹çš„åç»­èŠ‚ç‚¹æ˜¯å–æ¶ˆçŠ¶æ€... éœ€è¦æ‰¾ä¸€ä¸ªåˆé€‚çš„å¯ä»¥è¢«å”¤é†’çš„èŠ‚ç‚¹ã€‚æ­¤å¤„å¯ä»¥ç”»å›¾çœ‹çœ‹ã€‚
    /*
            headæŒ‡å‘èŠ‚ç‚¹0ï¼ŒsæŒ‡å‘èŠ‚ç‚¹1ï¼ŒtæŒ‡å‘èŠ‚ç‚¹5ã€‚ åˆ™æœ€ç»ˆsæŒ‡å‘èŠ‚ç‚¹3
             0       1      2       3       4      5
            node -> [1] -> [1] -> [-1] -> [-1] -> [0]
         */
    Node s = node.next;
    if (s == null || s.waitStatus > 0) {
        // æŸ¥æ‰¾å¯ä»¥è¢«å”¤é†’çš„èŠ‚ç‚¹...
        s = null;
        for (Node t = tail; t != null && t != node; t = t.prev)
            if (t.waitStatus <= 0)
                s = t;
        // ä¸Šé¢å¾ªç¯ï¼Œä¼šæ‰¾åˆ°ä¸€ä¸ªç¦»å½“å‰nodeæœ€è¿‘çš„ä¸€ä¸ªå¯ä»¥è¢«å”¤é†’çš„nodeï¼Œnodeå¯èƒ½æ‰¾ä¸åˆ°ï¼Œnodeå¯èƒ½æ˜¯null
    }
    // å¦‚æœæ‰¾åˆ°åˆé€‚çš„å¯ä»¥è¢«å”¤é†’çš„nodeï¼Œåˆ™å”¤é†’..ï¼Œæ‰¾ä¸åˆ°å•¥ä¹Ÿä¸åš
    if (s != null)
        LockSupport.unpark(s.thread);
}
